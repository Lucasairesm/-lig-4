import { TranslateModule,TranslateService  } from '@ngx-translate/core';
import { Component, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { RouterModule } from '@angular/router';
import { HttpClientModule } from '@angular/common/http';
import { TabViewModule } from 'primeng/tabview';
import { HomeComponent } from "./components/home/home.component";


@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.scss'],
  standalone: true,
  imports: [
    CommonModule,
    RouterModule,
    HttpClientModule,
    TranslateModule,
    TabViewModule,
    HomeComponent
]
})
export class AppComponent implements OnInit {

  constructor(
    private translateService: TranslateService
  ) {}

  setLanguage(lang: string): void {
    this.translateService.use(lang);
  }

  // Recupera o idioma salvo do localStorage
  ngOnInit(): void {
    this.translateService.use('pt-BR'); // Padrão inicial
  }
}

          <li>Os jogadores se alternam com um movimento de cada jogador, inserindo uma peça de sua cor em qualquer

          <li>A peça cai na posição mais baixa disponível da coluna e cada peça seguinte vai preenchendo os
              espaços vazios ou ficando sobre a peça abaixo.</li>
          <li>O jogo continua até que um dos jogadores consiga alinhar quatro peças consecutivas, ou até que o
              tabuleiro fique completamente cheio, resultando em um empate.</li>

      <p>O jogo acaba após um dos jogadores conseguir alinhar as 4 peças, dessa forma é declarado o vencedor por
          meio da cor escolhida pelo jogador. No fim, o jogador deve clicar no botão para encerrar a partida e
          iniciar um novo jogo.</p>
      <p>Em caso de empate, os jogadores terão a escolha de reiniciar a partida ou sair do jogo.</p>

      <h3>Visão Geral do Projeto</h3>
      <p>O projeto de desenvolvimento do jogo Ligue 4 para web visa criar uma versão online do clássico jogo de
          tabuleiro, permitindo que jogadores disputem partidas entre si em um ambiente digital.</p>

      <h3>Objetivos do Projeto</h3>
      <ul>
          <li>Desenvolver uma versão online do jogo Ligue 4 para rodar em navegadores web.</li>
          <li>Permitir que jogadores humanos possam competir entre si em tempo real.</li>
          <li>Garantir uma interface intuitiva e responsiva para uma experiência de jogo agradável.</li>
      </ul>

      <h3>Escopo do Projeto</h3>
      <p>O escopo do projeto inclui:</p>
      <ul>
          <li>Desenvolvimento do front-end e back-end necessários para o funcionamento do jogo.</li>
          <li>Implementação de um sistema de gerenciamento de partidas entre jogadores.</li>
